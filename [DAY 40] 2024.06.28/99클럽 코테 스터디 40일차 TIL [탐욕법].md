class Solution:
    def maximumOddBinaryNumber(self, s: str) -> str:
        sorted_s = sorted(s, reverse=True)
        for i in range(len(s) - 1, -1, -1):
            if sorted_s[i] == '1':
                sorted_s[i], sorted_s[-1] = sorted_s[-1], sorted_s[i]
                break
        return ''.join(sorted_s)

class Solution:
    def maximumOddBinaryNumber(self, s: str) -> str:
        cnt1 = s.count('1')
        cnt0 = s.count('0')
        ans = '1' * (cnt1 - 1) + '0' * cnt0 + '1'
        return ans
