class RecentCounter:

    def __init__(self):
        self.queue = deque()

    def ping(self, t: int) -> int:
        self.queue.append(t)
        while self.queue[0] < t-3000:

            self.queue.popleft()

    def ping(self, t: int) -> int:

        while t - self.q[0] > 3000:
            self.q.popleft()
