```python
class Solution:
    def countBits(self, n: int) -> List[int]:
        second = 0
        answer = []
        if not n:
            return [0]
        for num in range(n+1):
            for i in range(num, -1, -1):
                if num//(2**i) > 0:
                    second += 1
                    num -= 2**i
            answer.append(second)
            second = 0

        return answer
```

```python
class Solution:

            answer.append(sum(map(int,bin(num)[2:])))
```

```python
class Solution:
    def countBits(self, num: int) -> List[int]:
        dp = [0]
        for i in range(1, num + 1):
            if i % 2 == 1:
                dp.append(dp[i - 1] + 1)
            else:
                dp.append(dp[i // 2])
        return dp
```
